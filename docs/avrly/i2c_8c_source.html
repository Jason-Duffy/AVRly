44void init_i2c(uint32_t bus_speed)
45{
46 uint32_t base_frequency = (F_CPU / bus_speed);
47 TWBR = (uint8_t)((base_frequency - 16) / (2 * 1)); // Prescaler of 1
48 TWCR |= (1 << TWEN); // Enable
49}
