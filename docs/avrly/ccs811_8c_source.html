32/**
33 * @file ccs811.c
34 * \ingroup ccs811
35 * @author Jason Duffy
36 * @date 4th March 2022
37 * @brief Driver for the CCS811 gas sensor.
38 * @bug No known bugs.
39 *
40 * This file provides the basic low-level functionality for the CCS81 gas
41 * sensor / air quality sensor. It was tested with the CJMCU-811 module which
42 * provides pin breakout to THT for breadboard prototyping, and some peripheral
43 * circuitry. The CCS811 eatures an on-board MCU to interface with, greatly
44 * reducing the load on the host MCU.
45 *
46 * When the sensor is new, a burn-in period of 48hr is necessary for the
47 * resistive elements to level out and make readings more accurate. The CCS81
48 * controls the burn-in period, allowing eCO2 and eTVOC readings to be used
49 * from first power-on after 60 minutes of operation in modes 1-3.
50 *
51 * After early-life (Burn-In) use the conditioning period is the time required
52 * to achieve good sensor stability before measuring VOCs after a long idle
53 * period. After starting the application and calling init_ccs811(), run the
54 * sensor for 20 minutes before accurate readings are generated. The
55 * conditioning period must also be observed before writing to the BASELINE
56 * register.
57 *
58 * This driver was written using the datasheet for the ams CCS81 gas sensor,
59 * which can be found at the link below.
60 * @see
61 * https://pdf1.alldatasheet.com/datasheet-pdf/view/1047395/AMSCO/CCS811.html
62 */
63

141uint8_t init_ccs811(ccs811_config_t *p_config)
142{
143 p_config_global = p_config; // Make a copy of pointer with file scope.
144 uint8_t ret = 0; // For return value
145 init_i2c(BUS_SPEED_100KHZ);
146 if (ccs811_read_byte(HW_ID_REG) == CCS811_ID)
147 {
148 uint8_t status = ccs811_get_status();
149 if (bit_is_set(status, APP_VALID_BIT))
150 {
151 i2c_start();
152 i2c_send(CCS811_ADDRESS_W);
153 i2c_send(APP_START);
154 i2c_stop();
155 _delay_ms(100);
156
157 uint8_t byte = 0;
158 byte |= (p_config_global->drive_mode << DRIVE_MODE_BITS);
159 byte |= (p_config_global->interrupt_dataready << INT_DATARDY_BIT);
160 byte |= (p_config_global->interrupt_threshold <<INT_THRESH_BIT);
161
162 for (uint8_t count = 0; count < ATTEMPTS_MAX; ++count)
163 {
164 ccs811_write_byte(MEAS_MODE_REG, byte);
165 _delay_ms(500);
166 uint8_t mode = ccs811_read_byte(MEAS_MODE_REG);
167 if (mode == byte)
168 {
169 break; // Success
170 }
171 if (count == (ATTEMPTS_MAX - 1))
172 {
173 ret = 3; // Max attempts reached
174 }
175 }
176 }
177 else
178 {
179 ret = 2; // Error - app invalid
180 }
181 }
182 else
183 {
184 ret = 1; // ccs81 ID failed
185 }
186 return ret;
187}

197uint16_t ccs811_get_etvoc_level(void)
198{
199 uint8_t data = 0;
200 ccs811_burst_read(ALG_RESULT_DATA_REG, ETVOC_SIZE);
201 uint8_t msb = read_buffer[ETVOC_SIZE - 2];
202 data = read_buffer[ETVOC_SIZE - 1];
203 data |= (msb << 8);
204 return data;
205}

224void ccs811_update_env_data(uint8_t humidity, uint8_t temp)
225{
226 uint8_t lsb = 0;
227
228 i2c_start();
229 i2c_send(CCS811_ADDRESS_W);
230 i2c_send(ENV_DATA_REG);
231 i2c_send(humidity << 1);
232 i2c_send(lsb);
233 i2c_send(((temp + 25) << 1));
234 i2c_send(lsb);
235 i2c_stop();
236}

239char* ccs811_error_to_string(void)
240{
241 char *msg;
242 uint8_t error = ccs811_read_byte(ERROR_ID_REG);
243 if (bit_is_set(error, HEATER_SUPPLY))
244 {
245 msg = "Heater Supply";
246 }
247 else if (bit_is_set(error, HEATER_FAULT))
248 {
249 msg = "Heater Fault";
250 }
251 else if (bit_is_set(error, MAX_RESISTANCE))
252 {
253 msg = "Max Resistance";
254 }
255 else if (bit_is_set(error, MEASMODE_INVALID))
256 {
257 msg = "Invalid Meas Mode";
258 }
259 else if (bit_is_set(error, READ_REG_INVALID))
260 {
261 msg = "Read Reg Invalid";
262 }
263 else if (bit_is_set(error, WRITE_REG_INVALID))
264 {
265 msg = "Write Reg Invalid";
266 }
267 else
268 {
269 msg = "Unknown Error";
270 }
271 return msg;
272}

308void ccs811_burst_read(uint8_t reg, uint8_t length)
309{
310 i2c_start();
311 i2c_send(CCS811_ADDRESS_W);
312 i2c_send(reg);
313 i2c_start();
314 i2c_send(CCS811_ADDRESS_R);
315
316 for (uint8_t i = 0; i < length; ++i)
317 {
318 read_buffer[i] = i2c_read_ack();
319 }
320 i2c_stop();
321}

323uint16_t ccs811_read_word(uint8_t reg)
324{
325 uint16_t msb;
326 uint16_t byte;
327
328 i2c_start();
329 i2c_send(CCS811_ADDRESS_W);
330 i2c_send(reg);
331 i2c_start();
332 i2c_send(CCS811_ADDRESS_R);
333
334 msb = i2c_read_ack();
335 byte = i2c_read_no_ack();
336 byte |= (msb << 8);
337
338 return byte;
339}
