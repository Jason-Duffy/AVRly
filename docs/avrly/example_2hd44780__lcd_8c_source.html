31/**
32 * @file hd44780_lcd.c
33 * \ingroup hd44780_lcd
34 * @author Jason Duffy
35 * @date 1st March 2022
36 * @brief Driver for the HD44780 based 16x2 liquid crystal display.
37 * @bug No known bugs.
38 *
39 * This file provides the basic low-level functionality for the ubiquitous 16x2
40 * display. Please note that it uses long blocking waits in the initialisation
41 * routine (100ms), and short blockign waits in other utility functions
42 * (2uS - 2mS). This was done to simplify the code as the busy flag read
43 * introduced pitfalls. This driver currently only allows for a single display
44 * to be used, but this might be improved on at a later date.
45 *
46 * NOTE: Values used in bitwise operations MUST be of unsigned type.
47 *
48 * This driver was written using the datasheet for the HITACHI HD44780U LCD
49 * driver chip, which can be found here:
50 * @see http://www.datasheet-pdf.com/PDF/HD44780U-Datasheet-Hitachi-1109874
51 */
52

66/**
67 * Bitmask used to manipulate data based on data length setting in config.
68 */
69static uint8_t data_length_bitmask = 0;

71// Base commands without flags applied.
72#define CLEAR_DISPLAY 0b00000001U
73#define CURSOR_HOME 0b00000010U
74#define ENTRY_MODE_SET 0b00000100U
75#define ON_OFF_CTRL 0b00001000U
76#define CURSOR_DISPLAY_SHIFT 0b00010000U
77#define FUNCTION_SET 0b00100000U
78
79// Bit positions for setting flags in base commands.
80#define DATA_LENGTH_BIT 4U
81#define DISPLAY_LINES_BIT 3U
82#define FONT_SIZE_BIT 2U
83#define MOVE_DIRECTION_BIT 1U
84#define DISPLAY_SHIFT_BIT 0U
85#define CURSOR_ENABLE_BIT 1U
86#define BLINK_ENABLE_BIT 0U
87#define ON_OFF_CTRL_BIT 2U
88#define RIGHT_LEFT_BIT 2U
89#define SHIFT_OR_CURSOR_BIT 3U
90#define BUSY_FLAG_BIT 7U // Not currently used in this program.
91
92// Flags for setting ON_OFF_CTRL_BIT.
93#define DISPLAY_ON true
94#define DISPLAY_OFF false

98#define SHORT_INSTR_DELAY_MS 2
99#define LONG_INSTR_DELAY_MS 10
100#define SCROLL_DELAY_MS 400
101#define ENABLE_DELAY_US 2

109/*
110 * Initialisation routine (run once at startup).
111 */
112void init_lcd(lcd_config_t *p_config)
113{
114 // copy address of config object pointer to a local pointer
115 p_config_local = p_config;
116
117 // Set data direction registers to output where applicable
118 LCD_CTRL_DDR |= (1U << LCD_RS) | (1U << LCD_EN);
119
120 // Test for bit mode and set appropriate bitmask
121 if (p_config_local->eight_bit_mode)
122 {
123 data_length_bitmask = 0xff; // 0b11111111
124 }
125 else
126 {
127 data_length_bitmask |= (1U << LCD_D7) | (1U << LCD_D6);
128 data_length_bitmask |= (1U << LCD_D5) | (1U << LCD_D4);
129 }
130
131 // Set relevant lcd data port direction to output
132 LCD_DATA_DDR |= data_length_bitmask;
133
134 // Delay to allow power ramp up in LCD controller
135 _delay_ms(POWER_RAMP_DELAY_MS);
136
137 // Begin initialising HD44780
138 uint8_t byte = 0;
139 byte |= FUNCTION_SET;
140 byte |= (p_config_local->eight_bit_mode << DATA_LENGTH_BIT);
141
142 for (uint8_t count = 0; count < 3; ++count)
143 {
144 lcd_command(byte);
145 _delay_ms(LONG_INSTR_DELAY_MS);
146 }
147
148 byte = 0;
149 byte |= FUNCTION_SET;
150 byte |= (p_config_local->eight_bit_mode << DATA_LENGTH_BIT);
151 byte |= (p_config_local->two_line_display << DISPLAY_LINES_BIT);
152 byte |= (p_config_local->five_by_ten_font << FONT_SIZE_BIT);
153 lcd_command(byte);
154
155 byte = 0;
156 byte |= ON_OFF_CTRL;
157 byte |= (DISPLAY_OFF << ON_OFF_CTRL_BIT);
158 byte |= (p_config_local->cursor_enable << CURSOR_ENABLE_BIT);
159 byte |= (p_config_local->blink_enable << BLINK_ENABLE_BIT);
160 lcd_command(byte);
161
162 byte = 0;
163 byte |= CLEAR_DISPLAY;
164 lcd_command(byte);
165
166 byte = 0;
167 byte |= ENTRY_MODE_SET;
168 byte |= (p_config_local->increment_counter << MOVE_DIRECTION_BIT);
169 byte |= (p_config_local->display_shift << DISPLAY_SHIFT_BIT);
170 lcd_command(byte);
171
172 byte = 0;
173 byte |= ON_OFF_CTRL;
174 byte |= (DISPLAY_ON << ON_OFF_CTRL_BIT);
175 byte |= (p_config_local->cursor_enable << CURSOR_ENABLE_BIT);
176 byte |= (p_config_local->blink_enable << BLINK_ENABLE_BIT);
177 lcd_command(byte);
178 lcd_return_home();
179}

182/*
183 * Turn display off (config settings are retained).
184 */
185void lcd_display_off(void)
186{
187 uint8_t byte = 0;
188 byte |= ON_OFF_CTRL; // Default is off
189 byte |= (p_config_local->cursor_enable << CURSOR_ENABLE_BIT);
190 byte |= (p_config_local->blink_enable << BLINK_ENABLE_BIT);
191 lcd_command(byte);
192}

202 byte |= (DISPLAY_ON << ON_OFF_CTRL_BIT);
203 byte |= (p_config_local->cursor_enable << CURSOR_ENABLE_BIT);

212void lcd_print_string(const char *str)

221/*
222 * Prints an integer variable.
223 */
224void lcd_print_integer(int16_t number)
225{
226 if (number < 0)
227 {
228 lcd_char('-');
229 }
230
231 uint8_t count = 0;
232 uint8_t digit = 0;
233 char str[20];
234
235 while (number > 0)
236 {
237 digit = number % 10;
238 str[count] = (digit + '0');
239 ++count;
240 number /= 10;
241 }
242
243 while (count > 0)
244 {
245 lcd_char(str[count - 1]);
246 --count;
247 }
248}

251/*
252 * Sets cursor location using x and y coordinates.
253 */
254void lcd_set_cursor(uint8_t column, uint8_t row)
255{
256 uint8_t address = 0;
257
258 if (row == 0)
259 {
260 address = column;
261 }
262
263 else if (row == 1)
264 {
265 address = (column + 64);
266 }
267
268 address |= 0x80U; // Write 1 to MSB
269 lcd_command(address);
270}

273/*
274 * Writes space characters to all 32 sections of display (or 16 if in 1 line
275 * mode).
276 */
277void lcd_fast_clear(void)
278{
279 lcd_set_cursor(0,0);
280 for (uint8_t count = 0; count < 16; ++count)
281 {
282 lcd_char(' ');
283 }
284
285 if (p_config_local->two_line_display)
286 {
287 lcd_set_cursor(0,1);
288 for (uint8_t count = 0; count < 16; ++count)
289 {
290 lcd_char(' ');
291 }
292 }
293}

296/*
297 * Edits config settings for lcd (lcd_config_t members must be changed first).
298 */
299void lcd_reconfigure(void)
300{
301 uint8_t byte = 0;
302 byte |= ENTRY_MODE_SET;
303 byte |= (p_config_local->increment_counter << MOVE_DIRECTION_BIT);
304 byte |= (p_config_local->display_shift << DISPLAY_SHIFT_BIT);
305 lcd_command(byte);
306
307 // Also resends cursor and blink settings
308 lcd_display_on();
309}

400/*
401 * Sends a command to the display.
402 */
403void lcd_command(uint8_t data)
404{
405 LCD_CTRL_PORT &= ~(1U << LCD_RS);
406
407 if (p_config_local->eight_bit_mode)
408 {
409 LCD_DATA_PORT = data;
410 pulse_enable();
411 }
412 else
413 {
414 uint8_t byte = 0;
415
416 LCD_DATA_PORT &= ~(data_length_bitmask); // Clear relevant bits in port
417 byte = (data & 0b11110000); // Keep high 4 bits only
418 byte = (byte >> (7U - LCD_D7)); // Shift nibble into place
419 LCD_DATA_PORT |= byte; // Send first nibble to LCD
420 pulse_enable();
421
422 LCD_DATA_PORT &= ~(data_length_bitmask); // Clear relevant bits in port
423 byte = (data & 0b00001111); // Keep low 4 bits only
424 byte = (byte << (LCD_D7 - 3U)); // Shift nibble into place
425 LCD_DATA_PORT |= byte; // Send second nibble to LCD
426 pulse_enable();
427 }
428 _delay_ms(SHORT_INSTR_DELAY_MS);
429}

432/*
433 * Writes a single character to LCD.
434 */
435void lcd_char(char character)
436{
437 LCD_CTRL_PORT |= (1U << LCD_RS);
438
439 if (p_config_local->eight_bit_mode)
440 {
441 LCD_DATA_PORT = character;
442 pulse_enable();
443 }
444 else
445 {
446 uint8_t byte = 0;
447
448 LCD_DATA_PORT &= ~(data_length_bitmask); // Clear relevant bits in port
449 byte = (character & 0b11110000); // Keep high 4 bits only
450 byte = (byte >> (7U - LCD_D7)); // Shift nibble into place
451 LCD_DATA_PORT |= byte; // Send first nibble to LCD
452 pulse_enable();
453
454 LCD_DATA_PORT &= ~(data_length_bitmask); // Clear relevant bits in port
455 byte = (character & 0b00001111); // Keep low 4 bits only
456 byte = (byte << (LCD_D7 - 3U)); // Shift nibble into place
457 LCD_DATA_PORT |= byte; // Send second nibble to LCD
458 pulse_enable();
459 }

464/*
465 * Pulse enable pin to latch data into register.
466 */
467void pulse_enable(void)
468{
469 LCD_CTRL_PORT |= (1U << LCD_EN);
470 _delay_us(ENABLE_DELAY_US);
471 LCD_CTRL_PORT &= ~(1U << LCD_EN);
472 _delay_us(ENABLE_DELAY_US);
473}
