71/*
72 * Initialisation routine - call this function once at startup before using
73 * other functions. Log system will then be turned on by default - call
74 * log_global_off() if you do not wish it to be.
75 */
76void init_log_system(void)
77{
78 init_usart();
79 log_global_on();
80 log_message(&log_system_log, INFO, "Log system initialised");
81};

84/*
85 * Sends only the system tag, log level and message string.
86 * @param p_config is a pointer to the log_system config object. Instantiate
87 * the config object at the head of each file where logging is required and
88 * pass it's address into this function.
89 * @param level is the level status of the log message - see log_type_t for
90 * available options.
91 * @param msg is the message to be logged, enclose it in "" quotation marks.
92 */
93void log_message(log_system_config_t *p_config,
94 log_type_t level,
95 const char *msg)
96{
97 // If all test expressions evaluate true, log message.
98 if (log_message_preference_check(p_config, level))
99 {
100 print_tag_and_log_level(p_config->p_system_tag, level);
101 usart_print_string(msg);
102 }
103}

232/**
233 * Utility function to print labels over serial.
234 */
235void print_tag_and_log_level(const char *p_tag, log_type_t level)
236{
237 if (level == NONE)
238 {
239
240 }
241 usart_print_string("\n");
242 usart_print_string(p_tag);
243
244 if (level == ERROR)
245 {
246 usart_print_string(", ERROR: ");
247 }
248
249 else if (level == WARNING)
250 {
251 usart_print_string(", WARNING: ");
252 }
253
254 else if (level == INFO)
255 {
256 usart_print_string(", INFO: ");
257 }
258
259 else if (level == DEBUG)
260 {
261 usart_print_string(", DEBUG: ");
262 }
263
264 else if (level == VERBOSE_DEBUG)
265 {
266 usart_print_string(", VERBOSE_DEBUG: ");
267 }
268
269 else
270 {
271 usart_print_string(", INVALID_LOG_LEVEL: ");
272 }
273}

276/**
277 * Utility function to test if the log message level meets the preferences set.
278 */
279bool log_message_preference_check(log_system_config_t *p_config,
280 log_type_t level)
281{
282 // Test if log_system is enabled globally, if not then do nothing.
283 if (log_system_enabled)
284 {
285 // Test if the level of this log message meets global log preferences.
286 if (level <= global_max_output_level)
287 {
288 // Test if the level of this log message meets file level preferences.
289 if (level <= p_config->file_log_level)
290 {
291 return true;
292 }
293 }
294 }
295 // If any of the above conditionals evaluate to false, return false.
296 return false;
297}
