32/**
33 * @file mcp4812_dac.c
34 * \ingroup mcp48x2
35 * @author Jason Duffy
36 * @date 4th March 2022
37 * @brief Driver for the MCP4812 10 bit DAC (digital to analog converter) chip.
38 * @bug No known bugs.
39 *
40 * This file provides the basic low-level functionality for the MCP4812 10 bit
41 * DAC from Microchip. This driver currently only allows for a single DAC to be
42 * used, but this might be improved on at a later date.
43 *
44 * This driver was written using the datasheet for the Microchip MCP4812 chip,
45 * which can be found at the link below.
46 * @see https://ww1.microchip.com/downloads/en/DeviceDoc/20002249B.pdf

74/**
75 * Initialisation routine (run once at startup).
76 * This function is to be called before using any other DAC functions.
77 * Instantiate the dac_config_t object first then pass it's address into and
78 * call init_lcd() before using any other lcd functions.
79 * @param p_config is a pointer to the dac_config_t object.
80 * @return Returns void.
81 */
82void init_dac(dac_config_t *p_config)
83{
84 p_config_global = p_config;
85/**
86 * Don't need to set phase & polarity as the default works with MCP4812 (0,0)
87 * or (1,1).
88 */
89 SPCR |= (1 << SPR0); // div 16, safer for breadboards
90 SPCR |= (1 << MSTR); // clockmaster
91 SPCR |= (1 << SPE); // enable
92
93 DAC_CTRL_DDR |= (1 << DAC_CS); // Set chip select as output.
94 DAC_CTRL_PORT |= (1 << DAC_CS); // Set CS line high (not selected).
95 DAC_CTRL_DDR |= (1 << LDAC); // Set LDAC line (latch) as output.
96 DAC_CTRL_PORT |= (1 << LDAC); // Set LDAC high.
97 SPI_DDR |= (1 << SPI_MOSI); // Set MOSI as output.
98 SPI_PORT |= (1 << SPI_MISO); // Set pullup on MISO.
99 SPI_DDR |= (1 << SPI_SCK); // Set SCK as output.
100
101 sei(); // Set interrupt enable flag
102
103 if (!p_config_global->sync_manually)
104 {
105 DAC_CTRL_PORT &= ~(1 << LDAC);
106 }
107 dac_reconfigure(); // Apply config settings.
108}

111/**
112 * Sends a new millivolts value to be output on DAC.
113 */
114void dac_set_voltage(bool channel_a, uint16_t millivolts)
115{
116 if (channel_a)
117 {
118 if (p_config_global->channel_a.gain_low)
119 {
120 // Divide mV by 2 bitwise style (faster).
121 p_config_global->channel_a.level = (millivolts >> 1);
122 }
123 else
124 {
125 // Divide mV by 4 bitwise style (faster).
126 p_config_global->channel_a.level = (millivolts >> 2);
127 }
128 }
129 else
130 {
131 if (p_config_global->channel_b.gain_low)
132 {
133 // Divide mV by 2 bitwise style (faster).
134 p_config_global->channel_b.level = (millivolts >> 1);
135 }
136 else
137 {
138 // Divide mV by 4 bitwise style (faster).
139 p_config_global->channel_b.level = (millivolts >> 2);
140 }
141 }
142 // Send new values to DAC
143 dac_reconfigure();
144}

147/**
148 * Applies new config settings. This function takes updated config settings for
149 * both channels of DAC, then re-sends data so that the new settings take
150 * effect.
151 */
152void dac_reconfigure(void)
153{
154 uint16_t channel_a_data = 0;
155 uint16_t channel_b_data = 0;
156
157 channel_a_data &= ~(1 << CHANNEL_BIT); // Clear channel bit (Channel A).
158 channel_a_data |= (p_config_global->channel_a.gain_low << GAIN_BIT);
159 channel_a_data |= (p_config_global->channel_a.active << SHUTDOWN_BIT);
160
161 channel_b_data |= (1 << CHANNEL_BIT); // Set channel bit (Channel B).
162 channel_b_data |= (p_config_global->channel_b.gain_low << GAIN_BIT);
163 channel_b_data |= (p_config_global->channel_b.active << SHUTDOWN_BIT);
164
165 if (p_config_global->model == mcp4802)
166 {
167 channel_a_data |= (p_config_global->channel_a.level << EIGHT_BIT_OFFSET);
168 channel_b_data |= (p_config_global->channel_b.level << EIGHT_BIT_OFFSET);
169 }
170 else if (p_config_global->model == mcp4812)
171 {
172 channel_a_data |= (p_config_global->channel_a.level << TEN_BIT_OFFSET);
173 channel_b_data |= (p_config_global->channel_b.level << TEN_BIT_OFFSET);
174 }
175 else
176 {
177 channel_a_data |= (p_config_global->channel_a.level << TWELVE_BIT_OFFSET);
178 channel_b_data |= (p_config_global->channel_b.level << TWELVE_BIT_OFFSET);
179 }
180
181 spi_trade_byte(channel_a_data);
182 spi_trade_byte(channel_b_data);
183
184 if (!p_config_global->sync_manually)
185 {
186 pulse_latch();
187 }
188}

191//TODO: Make this interrupt based with SPIE bit in SPCR register (Pg198)
192void spi_trade_byte(uint16_t data)
193{
194 uint8_t msb = 0;
195 uint8_t lsb = 0;
196
197 chip_select();
198
199 msb |= (data >> 8U);
200 lsb |= (data);
201 SPDR = msb;
202 loop_until_bit_is_set(SPSR, SPIF);
203 SPDR = lsb;
204 loop_until_bit_is_set(SPSR, SPIF);
205
206 chip_deselect();
207}

209void pulse_latch(void)
210{
211 DAC_CTRL_PORT &= ~(1 << LDAC);
212 _delay_us(1);
213 DAC_CTRL_PORT |= (1 << LDAC);
214}
215
216void chip_select(void)
217{
218 DAC_CTRL_PORT &= ~(1 << DAC_CS);
219}
220
221
222void chip_deselect(void)
223{
224 DAC_CTRL_PORT |= (1 << DAC_CS);
225}
