42/**
43 * SCL frequency = F_CPU / (16 + (2 * TWBR) * PRESCALER)
44 */
45void init_i2c(uint32_t bus_speed)
46{
47 uint32_t base_frequency = (F_CPU / bus_speed);
48 TWBR = (uint8_t)((base_frequency - 16) / (2 * 1)); // Prescaler of 1
49 TWCR |= (1 << TWEN); // Enable
50}
51
52void i2c_wait_for_complete(void)
53{
54 loop_until_bit_is_set(TWCR, TWINT);
55}
56
57void i2c_start(void)
58{
59 TWCR = (_BV(TWINT) | _BV(TWEN) | _BV(TWSTA));
60 i2c_wait_for_complete();
61}
62
63void i2c_stop(void)
64{
65 TWCR = (_BV(TWINT) | _BV(TWEN) | _BV(TWSTO));
66}
67
68void i2c_send(uint8_t data)
69{
70 TWDR = data;
71 TWCR = (_BV(TWINT) | _BV(TWEN)); /* init and enable */
72 i2c_wait_for_complete();
73}
74
75
76uint8_t i2c_read_ack(void)
77{
78 TWCR = (_BV(TWINT) | _BV(TWEN) | _BV(TWEA));
79 i2c_wait_for_complete();
80 return (TWDR);
81}
82
83uint8_t i2c_read_no_ack(void)
84{
85 TWCR = (_BV(TWINT) | _BV(TWEN));
86 i2c_wait_for_complete();
87 return (TWDR);
88}
