29 * @bug No known bugs.
30 * @details This file provides the basic low-level functionality for the

33 * driver was written using the datasheet for the Microchip MCP4812 chip, which

45#define CHANNEL_BIT 15U
46#define GAIN_BIT 13U
47#define SHUTDOWN_BIT 12U
48#define TWELVE_BIT_LEVEL_OFFSET 0U

79/**
80 * Initialisation routine (run once at startup).
81 * This function is to be called before using any other DAC functions.
82 * Instantiate the dac_config_t object first then pass it's address into and
83 * call init_lcd() before using any other lcd functions.
84 * @param *p_config is a pointer to the dac_config_t object.
85 */
86void init_dac(dac_config_t *p_config)
87{
88 p_config_global = p_config;
89
90 // Delay to allow power ramp up in device.
91 _delay_ms(500);
92
93 init_spi(lsb_first,
94 controller,
95 rising_edge,
96 sample_leading_edge,
97 cpu_clk_div_16,
98 single_speed);
99
100 DAC_CTRL_DDR |= (1 << DAC_CS); // Set chip select as output.
101 DAC_CTRL_PORT |= (1 << DAC_CS); // Set CS line high (not selected).
102 DAC_CTRL_DDR |= (1 << LDAC); // Set LDAC line (latch) as output.
103 DAC_CTRL_PORT |= (1 << LDAC); // Set LDAC high.
104
105 // Save the value to bitshift millivolts value by.
106 if (p_config_global->model == mcp4802)
107 {
108 mv_resolution_shift = EIGHT_BIT_MV_OFFSET;
109 level_resolution_shift = EIGHT_BIT_LEVEL_OFFSET;
110 }
111 else if (p_config_global->model == mcp4812)
112 {
113 mv_resolution_shift = TEN_BIT_MV_OFFSET;
114 level_resolution_shift = TEN_BIT_LEVEL_OFFSET;
115 }
116 else
117 {
118 mv_resolution_shift = TWELVE_BIT_MV_OFFSET;
119 level_resolution_shift = TWELVE_BIT_LEVEL_OFFSET;
120 }
121
122 // Establish latching settings
123 if (!p_config_global->sync_manually)
124 {
125 DAC_CTRL_PORT &= ~(1 << LDAC);
126 }
127
128 // Send new values to DAC
129 dac_reconfigure();
130}

133/*
134 * Sends a new millivolts value to be output on DAC. Use this function got the
135 * MCP4802 and MCP4812 models.
136 */
137void dac_set_voltage(bool channel_a, uint16_t millivolts)
138{
139 // Manipulate mv value to suit register size of chip and gain setting.
140 if (channel_a)
141 {
142 if (p_config_global->channel_a.gain_low)
143 {
144 p_config_global->channel_a.level =
145 (millivolts >> mv_resolution_shift);
146 }
147 else
148 {
149 p_config_global->channel_a.level =
150 (millivolts >> (mv_resolution_shift + 1));
151 }
152 }
153 else
154 {
155 if (p_config_global->channel_b.gain_low)
156 {
157 p_config_global->channel_b.level =
158 (millivolts >> mv_resolution_shift);
159 }
160 else
161 {
162 p_config_global->channel_b.level =
163 (millivolts >> (mv_resolution_shift + 1));
164 }
165 }
166
167 // Send new values to DAC
168 dac_reconfigure();
169}

172/*
173 * Sends a new millivolts value to be output on DAC. Use this function for the
174 * MCP4822 model with 12bit resolution - for the other chip models use
175 * dac_set_voltage(). If gain_low = true, the resolution is down to 0.5mV,
176 * however we want to avoid using floating point numbers, so just pass in the
177 * integer millivolt value as the second parameter, then add a boolean
178 * true/false for the third parameter. True = 0.5mV, false = 0mV.
179 */
180void dac_set_voltage_12_bit(bool channel_a,
181 uint16_t millivolts,
182 bool fractional)
183{
184 if (channel_a)
185 {
186 if(p_config_global->channel_a.gain_low)
187 {
188 p_config_global->channel_a.level =
189 (millivolts << (TWELVE_BIT_MV_OFFSET + 1));
190 // Add the 0.5mV value if required.
191 if (fractional == true)
192 {
193 p_config_global->channel_a.level += 1;
194 }
195 }
196 else
197 {
198 p_config_global->channel_a.level =
199 (millivolts << TWELVE_BIT_MV_OFFSET);
200 }
201 }
202 else
203 {
204 if(p_config_global->channel_b.gain_low)
205 {
206 p_config_global->channel_b.level =
207 (millivolts << (TWELVE_BIT_MV_OFFSET + 1));
208 // Add the 0.5mV value if required.
209 if (fractional == true)
210 {
211 p_config_global->channel_b.level += 1;
212 }
213 }
214 else
215 {
216 p_config_global->channel_b.level =
217 (millivolts << TWELVE_BIT_MV_OFFSET);
218 }
219 }
220
221 // Send new values to DAC
222 dac_reconfigure();
223}

226/**
227 * Applies new config settings. This function takes updated config settings for
228 * both channels of DAC, then re-sends data so that the new settings take
229 * effect.
230 */
231void dac_reconfigure(void)
232{
233 uint16_t channel_a_data = 0;
234 uint16_t channel_b_data = 0;
235
236 // Set up config bits in 16bit word to be sent.
237 channel_a_data &= ~(1 << CHANNEL_BIT); // Clear channel bit (Channel A).
238 channel_a_data |= (p_config_global->channel_a.gain_low << GAIN_BIT);
239 channel_a_data |= (p_config_global->channel_a.active << SHUTDOWN_BIT);
240
241 channel_b_data |= (1 << CHANNEL_BIT); // Set channel bit (Channel B).
242 channel_b_data |= (p_config_global->channel_b.gain_low << GAIN_BIT);
243 channel_b_data |= (p_config_global->channel_b.active << SHUTDOWN_BIT);
244
245 // Shift level into correct place for chip and OR it into our 16bit word.
246 channel_a_data |=
247 (p_config_global->channel_a.level << level_resolution_shift);
248
249 channel_b_data |=
250 (p_config_global->channel_b.level << level_resolution_shift);
251
252 // Send new data to DAC over SPI.
253 chip_select();
254 spi_trade_word(channel_a_data);
255 chip_deselect();
256
257 chip_select();
258 spi_trade_word(channel_b_data);
259 chip_deselect();
260
261 // If sync_manually is set to false, pulse LDAC to update values.
262 if (!p_config_global->sync_manually)
263 {
264 pulse_latch();
265 }
266}

269/*
270 * Pulses LDAC pin pow for 1 microsecond. If sync_manually = true, call this
271 * function to latch the new voltage values into the output registers. If
272 * sync_manually = false, this function is called automatically by
273 * dac_reconfigure() or dac_set_voltage().
274 */
275void pulse_latch(void)
276{
277 DAC_CTRL_PORT &= ~(1 << LDAC);
278 _delay_us(1);
279 DAC_CTRL_PORT |= (1 << LDAC);
280}

287/*
288 * Private helper function - pulls CS line (Chip Select) low in order to begin
289 * SPI communication with DAC.
290 */
291void chip_select(void)
292{
293 DAC_CTRL_PORT &= ~(1 << DAC_CS);
294}
295
296/*
297 * Private helper function - pulls CS line (Chip Select) high to signal the end
298 * of SPI communication with DAC.
299 */
300void chip_deselect(void)
301{
302 DAC_CTRL_PORT |= (1 << DAC_CS);
303}
