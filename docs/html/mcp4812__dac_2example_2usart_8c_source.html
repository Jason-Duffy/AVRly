19/*
20 * Takes the defined BAUD and F_CPU, calculates the bit-clock multiplier,
21 * configures the hardware USART.
22 */
23void init_usart(void)
24{
25 // defined in setbaud.h
26 UBRR0H = UBRRH_VALUE;
27 UBRR0L = UBRRL_VALUE;
28
29#if USE_2X
30 UCSR0A |= (1 << U2X0);
31#else
32 UCSR0A &= ~(1 << U2X0);
33#endif
34
35 // Enable USART transmitter/receiver
36 UCSR0B = (1 << TXEN0) | (1 << RXEN0);
37 // 8 data bits, 1 stop bit
38 UCSR0C = (1 << UCSZ01) | (1 << UCSZ00);
39}

41// Utility function to transmit an entire string from RAM.
42void print_string(const char myString[])

48 i++;

52/*
53 * Define a string variable, pass it to this function. The string will contain
54 * whatever you typed over serial.
55 */
56void read_string(char myString[], uint8_t maxLength)
57{
58 char response;
59 uint8_t count = 0;
60
61 while (count < (maxLength - 1))
62 {
63 response = receive_byte();
64 transmit_byte(response); // echo
65
66 if (response == '\r') // enter marks the end
67 {
68 break;
69 }
70
71 else
72 {
73 myString[count] = response; // add in a letter
74 ++count;
75 }
76 }
77 myString[count] = 0; // terminal NULL character */
78}

80// Prints a byte out as its 3-digit ascii equivalent.
81void print_byte(uint8_t byte)
82{
83 // Converts a byte to a string of decimal text, sends it
84 transmit_byte('0' + (byte / 100)); // Hundreds
85 transmit_byte('0' + ((byte / 10) % 10)); // Tens
86 transmit_byte('0' + (byte % 10)); // Ones

89// Prints a word (16-bits) out as its 5-digit ascii equivalent.
90void print_word(uint16_t word)
91{
92 transmit_byte('0' + (word / 10000)); // Ten-thousands
93 transmit_byte('0' + ((word / 1000) % 10)); // Thousands
94 transmit_byte('0' + ((word / 100) % 10)); // Hundreds
95 transmit_byte('0' + ((word / 10) % 10)); // Tens
96 transmit_byte('0' + (word % 10)); // Ones
97}

99// Prints a byte out in 1s and 0s.
100void print_binary_byte(uint8_t byte)
101{
102 uint8_t bit;
103 for (bit = 7; bit < 255; bit--)
104 {
105 if (bit_is_set(byte, bit))
106 {
107 transmit_byte('1');
108 }
109 else
110 {
111 transmit_byte('0');
112 }
113 }
114}

116// Converts a nibble to a hexadecimal character

119 if (nibble < 10)

129// Prints a byte as its hexadecimal equivalent
130void print_hex_byte(uint8_t byte)
131{
132 uint8_t nibble;
133 nibble = (byte & 0b11110000) >> 4;
134 transmit_byte(nibble_to_hex_character(nibble));
135 nibble = byte & 0b00001111;
136 transmit_byte(nibble_to_hex_character(nibble));

139 // Gets a numerical 0-255 from the serial port, converts string to number.
140uint8_t get_number(void)
141{
142 char hundreds = '0';
143 char tens = '0';
144 char ones = '0';
145 char thisChar = '0';
146 do
147 {
148 hundreds = tens;
149 tens = ones;
150 ones = thisChar;
151 thisChar = receive_byte(); // get a new character
152 transmit_byte(thisChar); // echo
153 }
154 while (thisChar != '\r'); // until type return
155
156 // TODO: Bracketise this further
157 return (100 * (hundreds - '0') + 10 * (tens - '0') + ones - '0');
158}

170static uint8_t receive_byte(void)
171{
172 // Wait for incoming data
173 loop_until_bit_is_set(UCSR0A, RXC0);
174 //return register value
175 return UDR0;
176}
