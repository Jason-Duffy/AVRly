53/**
54 * Call this function once on entry of main() routine to configure settings and
55 * initialise the LED for use.
56 */
57void init_led(void)
58{
59 LED_DDR |= (1 << LED_GPIO); // Set LED_GPIO as output
60}

63/**
64 * Call this function in the loop of the main() routine to blink the LED.
65 */
66void blink_led(void)
67{
68 LED_PORT |= (1 << LED_GPIO); // Turn LED on
69 _delay_ms(BLINK_TIME_MS); // Wait a specified length of time
70
71 LED_PORT &= ~(1 << LED_GPIO); // Turn LED off
72 _delay_ms(BLINK_TIME_MS); // Wait a specified length of time
73}
