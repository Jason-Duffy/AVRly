23/**
24 * @file mcp4812_dac.h
25 * \ingroup mcp48x2
26 * @author Jason Duffy
27 * @date 4th March 2022
28 * @brief Driver for the MCP4812 10 bit DAC (digital to analog converter) chip.
29 * @bug No known bugs.
30 *
31 * This file provides the basic low-level functionality for the MCP4812 10 bit
32 * DAC from Microchip. This driver currently only allows for a single DAC to be
33 * used, but this might be improved on at a later date.
34 *
35 * This driver was written using the datasheet for the Microchip MCP4812 chip,
36 * which can be found at the link below.
37 * @see https://ww1.microchip.com/downloads/en/DeviceDoc/20002249B.pdf
38 */
39
40#ifndef MCP_FOUR_EIGHT_ONE_TWO_DOT_H
41#define MCP_FOUR_EIGHT_ONE_TWO_DOT_H
42
43#include <stdbool.h>
44#include <stdint.h>
45
46#define DAC_CHANNEL_A true
47#define DAC_CHANNEL_B false
48
49typedef enum
50{
51 mcp4802,
52 mcp4812,
53 mcp4822
54} model_num_t;
55
56typedef struct
57{
58 bool gain_low; // true = LOW (0 - 2048mV), false = HIGH (0 - 4096mV).
59 bool active; // true = Channel active, false = channel shutdown.
60 uint16_t level; // Value to be output on channel A.
61} dac_channel_t;
62
63typedef struct
64{
65 model_num_t model; // mcp4802, mcp4812 or mcp4822.
66 bool sync_manually; // true = manual sync, false = sync automatically.
67 dac_channel_t channel_a; // Config data members for channel A.
68 dac_channel_t channel_b; // Config data members for channel B.
69} dac_config_t;
70
71
72/**
73 * Initialisation routine (run once at startup).
74 * This function is to be called before using any other DAC functions.
75 * Instantiate the dac_config_t object first then pass it's address into and
76 * call init_lcd() before using any other lcd functions.
77 * @param p_config is a pointer to the dac_config_t object.
78 * @return Returns void.
79 */
80void init_dac(dac_config_t *p_config);
81
82
83/**
84 * Sends a new millivolts value to be output on DAC (Along with config settings).
85 */
86void dac_set_voltage(bool channel_a, uint16_t millivolts);
87
88
89/**
90 * Applies new config settings. This function takes updated config settings for
91 * both channels of DAC, then re-sends data so that the new settings take
92 * effect.
93 */
94void dac_reconfigure(void);
95
96
97/**
98 * Pulses LDAC pin low for a brief time (1uS). When sync_on_recofigure is
99 * false, and sync_manually is true, call this function to latch both config
100 * settings and new voltage value into both channels of DAC simultaneously.
101 */
102void pulse_latch(void);
