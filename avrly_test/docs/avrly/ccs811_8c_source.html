131uint8_t init_ccs811(ccs811_config_t *p_config)
132{
133 p_config_global = p_config; // Make a copy of pointer with file scope.
134 uint8_t ret = 0; // For return value
135 init_i2c(BUS_SPEED_100KHZ);
136 if (ccs811_read_byte(HW_ID_REG) == CCS811_ID)
137 {
138 uint8_t status = ccs811_get_status();
139 if (bit_is_set(status, APP_VALID_BIT))
140 {
141 i2c_start();
142 i2c_send(CCS811_ADDRESS_W);
143 i2c_send(APP_START);
144 i2c_stop();
145 _delay_ms(100);
146
147 uint8_t byte = 0;
148 byte |= (p_config_global->drive_mode << DRIVE_MODE_BITS);
149 byte |= (p_config_global->interrupt_dataready << INT_DATARDY_BIT);
150 byte |= (p_config_global->interrupt_threshold <<INT_THRESH_BIT);
151
152 for (uint8_t count = 0; count < ATTEMPTS_MAX; ++count)
153 {
154 ccs811_write_byte(MEAS_MODE_REG, byte);
155 _delay_ms(500);
156 uint8_t mode = ccs811_read_byte(MEAS_MODE_REG);
157 if (mode == byte)
158 {
159 break; // Success
160 }
161 if (count == (ATTEMPTS_MAX - 1))
162 {
163 ret = 3; // Max attempts reached
164 }
165 }
166 }
167 else
168 {
169 ret = 2; // Error - app invalid
170 }
171 }
172 else
173 {
174 ret = 1; // ccs81 ID failed
175 }
176 return ret;
177}

229char* ccs811_error_to_string(void)
230{
231 char *msg;
232 uint8_t error = ccs811_read_byte(ERROR_ID_REG);
233 if (bit_is_set(error, HEATER_SUPPLY))
234 {
235 msg = "Heater Supply";
236 }
237 else if (bit_is_set(error, HEATER_FAULT))
238 {
239 msg = "Heater Fault";
240 }
241 else if (bit_is_set(error, MAX_RESISTANCE))
242 {
243 msg = "Max Resistance";
244 }
245 else if (bit_is_set(error, MEASMODE_INVALID))
246 {
247 msg = "Invalid Meas Mode";
248 }
249 else if (bit_is_set(error, READ_REG_INVALID))
250 {
251 msg = "Read Reg Invalid";
252 }
253 else if (bit_is_set(error, WRITE_REG_INVALID))
254 {
255 msg = "Write Reg Invalid";
256 }
257 else
258 {
259 msg = "Unknown Error";
260 }
261 return msg;
